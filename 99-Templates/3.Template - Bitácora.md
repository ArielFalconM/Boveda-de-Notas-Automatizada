// 1- IDENTIFICACIÓN DEL PROYECTO
const currentFile = tp.config.target_file;
let nombreProyecto = "";

await new Promise(r => setTimeout(r, 300));
const recentFiles = app.workspace.getLastOpenFiles();

    const lastFile = app.vault.getAbstractFileByPath(recentFiles[1]);
    if (lastFile && lastFile.basename !== currentFile.basename) {
        nombreProyecto = lastFile.basename;
    }
}

// Si el historial falla, busca la nota del proyecto mediante backlinks
if (!nombreProyecto) {
    const backlinks = app.metadataCache.getBacklinksForFile(currentFile);
    if (backlinks && backlinks.data) {
        const paths = Object.keys(backlinks.data);
        if (paths.length > 0) {
            const sourceFile = app.vault.getAbstractFileByPath(paths[0]);
            if (sourceFile) nombreProyecto = sourceFile.basename;
        }
    }
}

// Si los dos mecanismos fallan, preguntar al usuario
if (!nombreProyecto || nombreProyecto === "Untitled") {
    nombreProyecto = await tp.system.prompt("Nombre del proyecto (Exacto):");
}


// 2-NOMBRES Y LINKS
let linkMapa = `[[Mapa - ${nombreProyecto}|🗺️ Ver Mapa del Proyecto]]`;

let proyectoLink = `[[${nombreProyecto}]]`;


// 3- SISTEMA DE HERENCIA DE TAGS
let archivoPadre = app.metadataCache.getFirstLinkpathDest(nombreProyecto, "");
let tagHeredado = "";

if (archivoPadre) {
    // Leemos el contenido de la nota del Proyecto
    let contenido = await app.vault.read(archivoPadre);

    // Buscamos de abajo hacia arriba la línea de Tags

            let primerTag = lineas[i].match(/(#[^\s#]+)/);
            if (primerTag) tagHeredado = primerTag[0];
            break;
        }
    }
}


// 4-OPERACIONES DE ARCHIVO

        const editor = leaf.view.editor;
        const lineas = editor.getValue().split('\n');
        
        // Foco directo en el objetivo de la sesión
        const index = lineas.findIndex(l => l.includes("Objetivo / Problema"));
        
        if (index !== -1) {
            editor.setCursor({ line: index + 1, ch: 0 });
            editor.focus();
        }
    }

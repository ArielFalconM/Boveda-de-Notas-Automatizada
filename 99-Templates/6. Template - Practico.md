<%*
// 1-IDENTIFICACIÓN DEL TEMA
const currentFile = tp.config.target_file;
let nombreTema = "";

await new Promise(r => setTimeout(r, 400));
const recentFiles = app.workspace.getLastOpenFiles();

if (recentFiles.length > 1) {
    const lastFile = app.vault.getAbstractFileByPath(recentFiles[1]);
    if (lastFile && lastFile.basename !== currentFile.basename) {
        nombreTema = lastFile.basename;
    }
}

if (!nombreTema || nombreTema === "Untitled") {
    nombreTema = await tp.system.prompt("¿A qué tema pertenece este práctico?");
}


// 2- ENTRADA DE DATOS DEL USUARIO
let nroPractico = await tp.system.prompt("Número o nombre de Práctico:");
let estados = ["⏳ Pendiente", "✏️ En proceso", "✔️ Completado"];
let estado_sel = await tp.system.suggester(estados, estados);

// Condicional para evitar pizarras sin utilizar.
let crearPizarra = await tp.system.suggester(["Sí, crear Pizarra de Resolución (Excalidraw)", "No, crear solo la nota de texto"], [true, false]);


// 3- NOMBRES Y RUTAS
const FILE_PRACTICO = `${nombreTema} - Práctico ${nroPractico}`;
const FILE_MAPA_PRACTICO = `Borrador - Práctico ${nroPractico} - ${nombreTema}`;
const FILE_MAPA_TEORIA = `Teoría - ${nombreTema}`; // Asume que el mapa teórico existe


// 4-SISTEMA DE HERENCIA DE TAGS
let archivoPadre = app.metadataCache.getFirstLinkpathDest(nombreTema, "");
let tagHeredado = "";

if (archivoPadre) {
    let contenido = await app.vault.read(archivoPadre);
    let lineas = contenido.split('\n');
    
    for (let i = lineas.length - 1; i >= 0; i--) {
        if (lineas[i].startsWith("Tags:")) {
            let primerTag = lineas[i].match(/(#[^\s#]+)/);
            if (primerTag) tagHeredado = primerTag[0];
            break;
        }
    }
}


// 5- OPERACIONES DE ARCHIVO Y RECURSOS

// Rename de la nota principal
await tp.file.rename(FILE_PRACTICO);

// Creación de la pizarra de práctico con inyección de tags
if (crearPizarra) {
    let t_excalidraw = app.metadataCache.getFirstLinkpathDest("Template-Excalidraw", "");
    let c_mapas = app.vault.getAbstractFileByPath("10-Zettelkasten");

    if (t_excalidraw && c_mapas) {
        let rutaDestinoMapa = `10-Zettelkasten/${FILE_MAPA_PRACTICO}.md`;
        let mapaExiste = app.vault.getAbstractFileByPath(rutaDestinoMapa);

        if (!mapaExiste) {
            try {

                let nuevoMapa = await app.vault.copy(t_excalidraw, rutaDestinoMapa);
                
                let contenidoMapa = await app.vault.read(nuevoMapa);
                await app.vault.modify(nuevoMapa, contenidoMapa + `\nTags: ${tagHeredado} #Excalidraw`);
                
                new Notice(`✅ Pizarra Práctica [${FILE_MAPA_PRACTICO}] preparada con herencia.`);
            } catch (e) {
                new Notice("❌ Error al copiar la pizarra. Revisa si el diagrama se creó.");
            }
        }
    } else {
        new Notice("❌ ERROR: Faltan recursos base (Template-Excalidraw o carpeta 10-Zettelkasten).");
    }
}


// 6- BYPASS DE FOCO
setTimeout(() => {
    const leaf = app.workspace.activeLeaf;
    if (leaf && leaf.view && leaf.view.editor) {
        const editor = leaf.view.editor;
        const lineas = editor.getValue().split('\n');
        
        // Busca específicamente el encabezado de esta template
        const index = lineas.findIndex(l => l.includes("Objetivos del Práctico"));
        
        if (index !== -1) {
            // Colocar el cursor exactamente en la línea de abajo y fuerza el foco
            editor.setCursor({ line: index + 1, ch: 0 });
            editor.focus();
        }
    }
}, 400);

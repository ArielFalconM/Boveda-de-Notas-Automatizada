<%*
// 1-ENTRADA DE DATOS Y ESTADO
let nombre = await tp.system.prompt("Nombre del proyecto:");
if (!nombre) nombre = "Proyecto_Sin_Nombre";

let tagProyecto = "#" + nombre.replace(/\s+/g, '');

let estados = ["💡 Idea", "⚙️ Prototipo", "💻 Desarrollo", "✅ Producción"];
let estado = await tp.system.suggester(estados, estados);

let numeroEstado = estados.indexOf(estado) + 1;
let estadoLimpio = estado ? estado.substring(3) : "Sin_Estado";
let carpetaEstado = estado ? `0${numeroEstado}-${estadoLimpio}` : "00-Sin_Estado";


// 2- CONFIGURACIÓN DE RECURSOS Y GITHUB
// Pregunta si crear el Mapa Excalidraw.
let crearMapa = await tp.system.suggester(["Sí, generar Mapa Excalidraw", "No, sin mapa"], [true, false]);

// Opción de GitHub
let opcionesGH = ["Sí, tendrá un repositorio asociado", "No, sin control de versiones"];
let seleccionGH = await tp.system.suggester(opcionesGH, opcionesGH);
let lineaRepo = "";

if (seleccionGH === opcionesGH[0]) {
    let link = await tp.system.prompt("Pega el enlace al repositorio (vacío para hacerlo luego):");
    if (!link) {
        lineaRepo = `\n**🔗 Repositorio:** [Link Pendiente](#) (Sin URL aún)`;
    } else {
        lineaRepo = `\n**🔗 Repositorio:** [Abrir en GitHub](${link})`;
    }
}


// 3- SELECCIÓN DE STACK TECNOLÓGICO
const archivosSoftware = app.vault.getMarkdownFiles().filter(archivo => {
    const cache = app.metadataCache.getFileCache(archivo);
    const tags = cache?.frontmatter?.tags || cache?.tags || [];
    return JSON.stringify(tags).includes("Software") && !archivo.path.includes("99-Templates");
});

let opcionesSoftware = archivosSoftware.map(f => f.basename).sort();
opcionesSoftware.unshift("➡️ Finalizar");

let stackSeleccionado = [];
while (true) {
    let sel = await tp.system.suggester(opcionesSoftware, opcionesSoftware);
    if (!sel || sel === "➡️ Finalizar") break; 
    stackSeleccionado.push(`[[${sel}]]`);
    opcionesSoftware = opcionesSoftware.filter(item => item !== sel);
    if (opcionesSoftware.length === 0) break;
}


// 4 - NOMBRES Y RUTAS
const nombreMapa = `Mapa - ${nombre}`;
const nombreKanban = `Kanban - ${nombre}`;
const FOLDER_PROYECTOS = "20-Proyectos";
const FOLDER_ZETTEL = "10-Zettelkasten";


// 5-OPERACIONES DE ARCHIVO Y RECURSOS
// Rename y move de la nota principal
await tp.file.rename(nombre);
await tp.file.move(`/${FOLDER_PROYECTOS}/${carpetaEstado}/${nombre}`);

// Clonación de Recursos (Excalidraw y Kanban)
let t_excalidraw = app.metadataCache.getFirstLinkpathDest("Template-Excalidraw", "");
let t_kanban = app.metadataCache.getFirstLinkpathDest("Template-Kanban-Software", "");

// Despliegue de Mapa (Condicional e inyección de tags)
if (crearMapa && t_excalidraw) {
    let rutaDestinoMapa = `${FOLDER_ZETTEL}/${nombreMapa}.md`;
    if (!app.vault.getAbstractFileByPath(rutaDestinoMapa)) {
        // Copia del recurso y la referencia
        let nuevoMapa = await app.vault.copy(t_excalidraw, rutaDestinoMapa);
        
        // Lectura de contenido e Inyección del tag del proyecto
        let contenidoMapa = await app.vault.read(nuevoMapa);
        await app.vault.modify(nuevoMapa, contenidoMapa + `\nTags: ${tagProyecto} #Excalidraw`);
        
        new Notice("✅ Mapa Excalidraw desplegado con herencia.");
    }
}

// Despliegue de Kanban (Con inyección de tags)
if (t_kanban) {
    let rutaDestinoKanban = `${FOLDER_ZETTEL}/${nombreKanban}.md`;
    if (!app.vault.getAbstractFileByPath(rutaDestinoKanban)) {
        // Copia del recurso y su referencia
        let nuevoKanban = await app.vault.copy(t_kanban, rutaDestinoKanban);
        
        // Inyección del tag del proyecto
        let contenidoKanban = await app.vault.read(nuevoKanban);
        await app.vault.modify(nuevoKanban, contenidoKanban + `\nTags: ${tagProyecto} #Kanban`);
        
        new Notice("✅ Tablero Kanban enviado a Zettelkasten con herencia.");
    }
} else {
    new Notice("❌ ERROR: No se encontró 'Template-Kanban-Software'.");
}


// 6-BYPASS DE FOCO
setTimeout(() => {
    const leaf = app.workspace.activeLeaf;
    if (leaf && leaf.view && leaf.view.editor) {
        const editor = leaf.view.editor;
        const lineas = editor.getValue().split('\n');
        
        // Busca específicamente el encabezado principal
        const index = lineas.findIndex(l => l.includes("Breakdown del Proyecto"));
        
        if (index !== -1) {
            // Coloca el cursor exactamente en la línea de abajo y fuerza el foco
            editor.setCursor({ line: index + 1, ch: 0 });
            editor.focus();
        }
    }
}, 400);
-%>

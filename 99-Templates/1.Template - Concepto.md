<%*
// 1-ENTRADA DE DATOS DEL USUARIO
let titulo = await tp.system.prompt("Nombre del concepto:");
if (!titulo) titulo = "Concepto_Sin_Titulo";

// 2- CONFIGURACIÓN DEL MENÚ DE ORIGEN
let linkPadre = "";
let textoOrigen = "Independiente";

// última nota abierta
let archivosRecientes = app.workspace.getLastOpenFiles();
let ultimaNota = (archivosRecientes.length > 1) ? app.vault.getAbstractFileByPath(archivosRecientes[1]) : null;

let opcionesMenu = [];
if (ultimaNota) {
    opcionesMenu.push(`📂 Última visitada: ${ultimaNota.basename}`);
}
opcionesMenu.push("🔍 Buscar por tipo (Tag)", "➡️ Saltar / Ninguno");

let seleccionMenu = await tp.system.suggester(opcionesMenu, opcionesMenu);


// 3-PROCESAMIENTO DEL ORIGEN
if (seleccionMenu && seleccionMenu.startsWith("📂")) {
    // Caso "Última nota visitada"
    linkPadre = `[[${ultimaNota.basename}]]`;
    textoOrigen = "Referencia Directa";
} 
else if (seleccionMenu === "🔍 Buscar por tipo (Tag)") {
    // Caso "Búsqueda manual por tags"
    let tiposOrigen = ["Tema", "Software", "Práctico", "Proyecto"];
    let tipoOrigen = await tp.system.suggester(tiposOrigen, tiposOrigen);
    
    if (tipoOrigen) {
        let tagBuscado = tipoOrigen.toLowerCase().replace("á", "a");
        const archivos = app.vault.getMarkdownFiles().filter(archivo => {
            const cache = app.metadataCache.getFileCache(archivo);
            const tags = cache?.frontmatter?.tags || cache?.tags || [];
            return JSON.stringify(tags).toLowerCase().includes(tagBuscado) && !archivo.path.includes("99-Templates");
        });

        let listaArchivos = archivos.map(f => f.basename).sort();
        if (listaArchivos.length > 0) {
            let seleccionFinal = await tp.system.suggester(listaArchivos, listaArchivos);
            if (seleccionFinal) {
                linkPadre = `[[${seleccionFinal}]]`;
                textoOrigen = tipoOrigen;
            }
        } else {
            new Notice(`⚠️ No hay notas con el tag #${tagBuscado}`);
        }
    }
}


// 4-SISTEMA DE HERENCIA DE TAGS
let archivoPadre = null;
let tagHeredado = "";

// Archivo físico padre
if (seleccionMenu && seleccionMenu.startsWith("📂")) {
    archivoPadre = ultimaNota;
} else if (linkPadre !== "") {
    let nombreLimpio = linkPadre.replace(/[[\]]/g, '');
    archivoPadre = app.metadataCache.getFirstLinkpathDest(nombreLimpio, "");
}

// Búsqueda de la tag padre en la nota padre ubicada en la ultima linea.
if (archivoPadre) {
    let contenido = await app.vault.read(archivoPadre);
    let lineas = contenido.split('\n');
    
    for (let i = lineas.length - 1; i >= 0; i--) {
        if (lineas[i].startsWith("Tags:")) {
            let primerTag = lineas[i].match(/(#[^\s#]+)/);
            if (primerTag) tagHeredado = primerTag[0];
            break;
        }
    }
}


// EXTRA

// Rename de nota
await tp.file.rename(titulo);

// Bypass: Forzar foco del cursor tras el renderizado de la UI
setTimeout(() => {
    const leaf = app.workspace.activeLeaf;
    if (leaf && leaf.view && leaf.view.editor) {
        const editor = leaf.view.editor;
        const lineas = editor.getValue().split('\n');
        
        // Busca específicamente el encabezado de esta template
        const index = lineas.findIndex(l => l.includes("Definición"));
        
        if (index !== -1) {
            // Coloca el cursor exactamente en la línea de abajo y fuerza el foco
            editor.setCursor({ line: index + 1, ch: 0 });
            editor.focus();
        }
    }
}, 400);
-%>

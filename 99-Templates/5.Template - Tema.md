<%*
// 1- ENTRADA DE DATOS DEL USUARIO
let titulo = await tp.system.prompt("Tema a estudiar:");
if (!titulo) titulo = "Tema_Sin_Nombre";

const FOLDER_MATERIAS = "97-Materias";
const FOLDER_ZETTEL = "10-Zettelkasten";


// 2-ELECCIÓN DINÁMICA DE MATERIA
const archivosMaterias = app.vault.getMarkdownFiles().filter(f => f.path.startsWith(FOLDER_MATERIAS + "/"));
let opcionesMaterias = archivosMaterias.map(f => f.basename);

if (opcionesMaterias.length === 0) opcionesMaterias = ["Sin_Materias_Registradas"];

let materia = await tp.system.suggester(opcionesMaterias, opcionesMaterias);
let tagMateria = "#" + (materia ? materia.replace(/\s+/g, '') : "SinMateria");


// 3-CONFIGURACIÓN DE RECURSOS
// Pregunta condicional para la pizarra de Excalidraw
let crearPizarra = await tp.system.suggester(["Sí, crear Pizarra Teórica (Excalidraw)", "No, crear solo la nota de texto"], [true, false]);

const FILE_TEMA = titulo;
const FILE_MAPA = `Teoría - ${titulo}`;


// 4-OPERACIONES DE ARCHIVO Y DESPLIEGUE

// Rename de la nota principal
await tp.file.rename(FILE_TEMA);

// Despliegue condicional de la pizarra con inyección de tags
if (crearPizarra) {
    let t_excalidraw = app.metadataCache.getFirstLinkpathDest("Template-Excalidraw", "");
    
    if (t_excalidraw) {
        let rutaDestinoMapa = `${FOLDER_ZETTEL}/${FILE_MAPA}.md`;
        let mapaExiste = app.vault.getAbstractFileByPath(rutaDestinoMapa);

        if (!mapaExiste) {
            try {
                let nuevoMapa = await app.vault.copy(t_excalidraw, rutaDestinoMapa);
                
                let contenidoMapa = await app.vault.read(nuevoMapa);
                await app.vault.modify(nuevoMapa, contenidoMapa + `\nTags: ${tagMateria} #Excalidraw`);
                
                new Notice(`✅ Pizarra [${FILE_MAPA}] preparada con herencia de ${tagMateria}`);
            } catch (error) {
                new Notice("❌ Error al crear la pizarra. Revisa si el diagrama se creó.");
            }
        }
    } else {
        new Notice("❌ ERROR: No se encontró 'Template-Excalidraw'.");
    }
}


// 5- BYPASS DE FOCO
setTimeout(() => {
    const leaf = app.workspace.activeLeaf;
    if (leaf && leaf.view && leaf.view.editor) {
        const editor = leaf.view.editor;
        const lineas = editor.getValue().split('\n');
        
        // Foco específico en la Idea Central del tema
        const index = lineas.findIndex(l => l.includes("Idea central"));
        
        if (index !== -1) {
            editor.setCursor({ line: index + 1, ch: 0 });
            editor.focus();
        }
    }
}, 400);
-%>
